library(leaflet)
## Warning: package 'leaflet' was built under R version 3.5.3
prefix = 'https://api.gbif.org/v2/map/occurrence/density/{z}/{x}/{y}@4x.png?'
style = 'style=classic.poly&bin=hex&hexPerTile=100'
taxonKey = 'taxonKey=2969235'
tile = paste0(prefix,style,'&',taxonKey)
leaflet() %>%
  setView(lng = 20, lat = 20, zoom = 02) %>%
  addTiles() %>%  
  addTiles(urlTemplate=tile)
